11/// An arena pre-allocates so-called *pages* of size Arena::page_size_.
12/// You can use Arena::allocate to obtain memory from this.
13/// When a page runs out of memory, the next page will be (pre-)allocated.
14/// You cannot directly release memory obtained via this method.
15/// Instead, *all* memory acquired via this Arena will be released as soon as this Arena will be destroyed.
16/// As an exception, you can Arena::deallocate memory that just as been acquired.
17class Arena {
18public:

50 /// @name Smart Pointer
51 ///@{
52 /// This is a [std::unique_ptr](https://en.cppreference.com/w/cpp/memory/unique_ptr)
53 /// that uses the Arena under the hood
54 /// and whose deleter will *only* invoke the destructor but *not* `delete` anything;
55 /// memory will be released upon destruction of the Arena.
56 ///
57 /// Use like this:
58 /// ```
59 /// auto ptr = arena.mk<Foo>(a, b, c); // new Foo(a, b, c) placed into arena
60 /// ```
61 template<class T> struct Deleter {
62 constexpr Deleter() noexcept = default;

65 };

69 auto ptr = new (allocate(sizeof(T))) T(std::forward<Args&&>(args)...);
70 return Ptr<T>(ptr, Deleter<T>());
71 }

87 /// Align next allocate(size_t) to @p a.
88 void align(size_t a) { index_ = (index_ + (a - 1)) & ~(a - 1); }
89
90 /// Get @p n bytes of fresh memory.
91 [[nodiscard]] void* allocate(size_t num_bytes) {
92 if (index_ + num_bytes > page_size_) {
93 pages_.emplace_back(new char[std::max(page_size_, num_bytes)]);
94 index_ = 0;
95 }
96
97 auto result = pages_.back() + index_;
98 index_ += num_bytes;
99 return result;
100 }
101
102 template<class T> [[nodiscard]] T* allocate(size_t num_elems) {
103 align(alignof(T));
104 return static_cast<T*>(allocate(num_elems * std::max(sizeof(T), alignof(T))));
105 }
106 ///@}
107
108 /// @name Dealloc
109 ///@{
110 /// Tries to remove bytes again.
111 /// If this crosses a page boundary, nothing happens.
112 /// Use like this:
113 /// ```
114 /// auto state = arena.state();
115 /// auto ptr = arena.allocate(n);
116 /// if (/* I don't want that */) arena.deallocate(state);
117 /// ```
118 /// @warning Only use, if you really know what you are doing.
119 using State = std::pair<size_t, size_t>;
120 State state() const { return {pages_.size(), index_}; }
121 void deallocate(State state) {
122 if (state.first == pages_.size()) index_ = state.second; // don't care otherwise
123 }
124 ///@}
